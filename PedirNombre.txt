// Función para pedir el nombre y saludar
function pedirNombre() {
  let nombre = prompt("Introduce tu nombre:");

  // Si el usuario no introduce un nombre, se usa "invitado"
  if (!nombre) {
    nombre = "invitado";
  }

  console.log(`Hola, ${nombre}!`);
}

// Llamar a la función para que pida el nombre
pedirNombre();
